            .data

mensagem:	.asciiz "Digite um numero\n"
num1: 	.word 1
dobro: 	.word 1
		
	.text
	# imprime mensagem
main:	li $v0, 4 
	la $a0, mensagem
	syscall

	# lê inteiro
	li $v0, 5 
	syscall
			
	# grava numero na memória	
	sw $v0, num1
			
	# calcular dobro
	mul $t0, $v0, $v0

	# grava dobro
	sw $t0, dobro
			
	# imprime inteiro
	li $v0, 1
	move $a0, $t0
	syscall

	jr $ra
